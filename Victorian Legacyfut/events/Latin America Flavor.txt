country_event = {

	id = 19007

	title = "Cores Removed"
	desc = ""
	picture = "theatre"

	trigger = {
		SAS = { 
			OR = {
				has_country_flag = brz_north_accepted
				has_country_flag = brz_south_accepted
			}
		}
		any_owned_province = { is_core = SAS }
		NOT = { vassal_of = SAS }
		NOT = { has_country_flag = cores_removed_SAS }
		NOT = { tag = SAS }
	}

	option = {
		name = "WE ARE FREE"
		set_country_flag = cores_removed_SAS
		any_owned = {
				limit = { is_core = SAS }
				remove_core = SAS
		}
	}
}
